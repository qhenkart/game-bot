from sikuli import *

quest = Region(50,219,266,294)
menu = Region(864,67,266,132)
list = Region(50,137,235,417)


dailyR = Region(309,518,1000,71)

def run():
    nox = App.open("Nox App Player")
    with Region(nox.window()):
        print("running weeklys")

        menu.click()
        list.wait()
        if list.exists():
            click()
            acceptDailys()

        list.click()
        wait(5)
        if exists():
            click()
            wait(2)
        if exists():
            click()
            wait(2)

        if exists():
           click()
        wait(2)
        lib.checkWalk()


        return;

def acceptDailys():
    nox = App.open("Nox App Player")
    with Region(nox.window()):
        print("checking dailys")


        while dailyR.exists():
            print("claiming dailys reward")
            dailyR.click()
            wait(2)
        while dailyR.exists():
            if exists():
                print("finished with quests")
                lib.checkOK()
                return;
            print("starting quest")
            dailyR.click()
            wait(2)
