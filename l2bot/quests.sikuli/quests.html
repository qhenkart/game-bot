from sikuli import *

import lib

quest = Region(49,266,272,112)

reward = Region(425,631,530,75)


cleared = Region(1025,658,199,66)

def runQuest():
    c = 1
    while not reward.exists():
        print("in loop" + str(c))
        c = c + 1
        lib.checkOK()
        lib.checkSkip()
        if quest.exists():
            print("clicking to accept quest")
            quest.click()
            lib.checkSkip()

        lib.checkWalk()
        print("checking accept")
        if exists():
            click()
            print("accepting quest")
        lib.checkSkip()

    return;


def run():
    if quest.exists():
        print("clicking to accept quest")
        quest.click()
        lib.checkSkip()
        #loops until claim reward is seen
        runQuest()

    #finish quest           
    if reward.exists():
        print("claiming reward")
        reward.click()
    #story progression
    if cleared.exists():
        print("story complete")
        cleared.click()

    return;
